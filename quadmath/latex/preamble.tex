% QuadMath LaTeX Preamble
% Optimized layout and formatting

% Enhanced geometry for minimal margins
\usepackage[
  margin=0.4cm,
  top=0.8cm,
  bottom=0.8cm,
  includeheadfoot
]{geometry}

% Improved spacing and typography
\usepackage{microtype}
\usepackage{setspace}
\setstretch{1.1}

% Better figure placement
\usepackage{float}
\floatplacement{figure}{H}
\floatplacement{table}{H}

% Compact lists
\usepackage{enumitem}
\setlist{nosep, leftmargin=1em}

% Enhanced math spacing
\usepackage{amsmath}
\allowdisplaybreaks

% Reduce space around section headers
\usepackage{titlesec}
\titlespacing*{\section}{0pt}{0.5ex plus 0.2ex minus 0.1ex}{0.3ex plus 0.1ex}
\titlespacing*{\subsection}{0pt}{0.4ex plus 0.15ex minus 0.1ex}{0.25ex plus 0.1ex}
\titlespacing*{\subsubsection}{0pt}{0.3ex plus 0.1ex minus 0.05ex}{0.2ex plus 0.05ex}

% Tighter paragraph spacing
\setlength{\parskip}{0.3ex plus 0.1ex minus 0.05ex}
\setlength{\parindent}{0pt}

% Optimize figure/table captions
\usepackage{caption}
\captionsetup{
  font=small,
  labelfont=bf,
  skip=0.2ex,
  belowskip=0.1ex
}

% Hyperlink colors (red)
\usepackage{xcolor}
\definecolor{linkred}{RGB}{220,20,60}
\hypersetup{
  colorlinks=true,
  linkcolor=linkred,
  urlcolor=linkred,
  citecolor=linkred
}
